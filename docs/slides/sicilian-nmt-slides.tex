\section{Data pipeline}
\begin{frame}{From PDFs to parallel sentences (all CPU)}
  \begin{enumerate}
    \item \textbf{Extract} (PyMuPDF): clean text, classify page language (even=scn/odd=en).
    \item \textbf{Align} (LaBSE + DP): confirm parallel pages, align sentences --- no
          dictionary, no hunalign.
    \item \textbf{Augment}: Napizia ``Good-Sicilian-in-NLLB'' (filtered), WikiMatrix it--scn.
    \item \textbf{Assemble}: dedup + split, literary held-out test set.
  \end{enumerate}

\section{Method}
\begin{frame}{Tokenization + desinence-biased subwords (``lever B'')}
  \begin{itemize}
    \item Pure-Python tokenizer: ASCII-fold + uncontract (\^o$\to$a lu, d\^u$\to$di lu).
    \item Bias joint BPE toward textbook desinences (Dieli / Chi\`u d\^a Palora).
    \item \textbf{+1.7 BLEU} over the raw baseline.
  \end{itemize}
\end{frame}

\begin{frame}{Models}
